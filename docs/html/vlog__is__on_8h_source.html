32// Defines the VLOG_IS_ON macro that controls the variable-verbosity
33// conditional logging.
34//
35// It's used by VLOG and VLOG_IF in logging.h
36// and by RAW_VLOG in raw_logging.h to trigger the logging.
37//
38// It can also be used directly e.g. like this:
39// if (VLOG_IS_ON(2)) {
40// // do some logging preparation and logging
41// // that can't be accomplished e.g. via just VLOG(2) << ...;
42// }
43//
44// The truth value that VLOG_IS_ON(level) returns is determined by
45// the three verbosity level flags:
46// --v=<n> Gives the default maximal active V-logging level;
47// 0 is the default.
48// Normally positive values are used for V-logging levels.
49// --vmodule=<str> Gives the per-module maximal V-logging levels to override
50// the value given by --v.
51// E.g. "my_module=2,foo*=3" would change the logging level
52// for all code in source files "my_module.*" and "foo*.*"
53// ("-inl" suffixes are also disregarded for this matching).
54//
55// SetVLOGLevel helper function is provided to do limited dynamic control over
56// V-logging by overriding the per-module settings given via --vmodule flag.
57//
58// CAVEAT: --vmodule functionality is not available in non gcc compilers.
59//
60
61#ifndef BASE_VLOG_IS_ON_H_

64#include <glog/log_severity.h>

66#if defined(__GNUC__)
67// We emit an anonymous static int* variable at every VLOG_IS_ON(n) site.
68// (Normally) the first time every VLOG_IS_ON(n) site is hit,
69// we determine what variable will dynamically control logging at this site:
70// it's either FLAGS_v or an appropriate internal variable
71// matching the current source file that represents results of
72// parsing of --vmodule flag and/or SetVLOGLevel calls.
73#define VLOG_IS_ON(verboselevel) \
74 __extension__ \
75 ({ static google::SiteFlag vlocal__ = {NULL, NULL, 0, NULL}; \
76 google::int32 verbose_level__ = (verboselevel); \
77 (vlocal__.level == NULL ? google::InitVLOG3__(&vlocal__, &FLAGS_v, \
78 __FILE__, verbose_level__) : *vlocal__.level >= verbose_level__); \
79 })
80#else
81// GNU extensions not available, so we do not support --vmodule.
82// Dynamic value of FLAGS_v always controls the logging level.
83#define VLOG_IS_ON(verboselevel) (FLAGS_v >= (verboselevel))
84#endif

86// Set VLOG(_IS_ON) level for module_pattern to log_level.
87// This lets us dynamically control what is normally set by the --vmodule flag.
88// Returns the level that previously applied to module_pattern.
89// NOTE: To change the log level for VLOG(_IS_ON) sites
90// that have already executed after/during InitGoogleLogging,
91// one needs to supply the exact --vmodule pattern that applied to them.
92// (If no --vmodule pattern applied to them
93// the value of FLAGS_v will continue to control them.)
94extern GLOG_EXPORT int SetVLOGLevel(const char* module_pattern, int log_level);
